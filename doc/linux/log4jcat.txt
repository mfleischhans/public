#!/usr/bin/env bash
# log4jcat - colorize Log4j-style plain text logs (no installs)
# Usage:
#   log4jcat file.log
#   cat file.log | log4jcat
#   log4jcat file.log | less -R

awk '
BEGIN {
  ESC = sprintf("%c", 27)
  reset = ESC "[0m"

  dim   = ESC "[90m"
  red   = ESC "[31m"
  bred  = ESC "[1;31m"
  yel   = ESC "[33m"
  grn   = ESC "[32m"
  cyn   = ESC "[36m"
  mag   = ESC "[35m"
  blu   = ESC "[34m"
  bblu  = ESC "[1;34m"
}

{
  line = $0

  # Stack traces / continuations
  if (line ~ /^[[:space:]]+at[[:space:]]/ ||
      line ~ /^[[:space:]]*\.\.\.[[:space:]]+[0-9]+[[:space:]]+more[[:space:]]*$/) {
    print dim line reset
    next
  }

  # "Caused by" / "Suppressed"
  if (line ~ /^[[:space:]]*Caused by:/) {
    print bred line reset
    next
  }
  if (line ~ /^[[:space:]]*Suppressed:/) {
    print yel line reset
    next
  }

  # Timestamp at beginning (common log4j/log4j2 patterns)
  if (match(line, /^([0-9]{4}-[0-9]{2}-[0-9]{2})([ T])([0-9]{2}:[0-9]{2}:[0-9]{2})([,.][0-9]+)?([Z]|[+-][0-9]{2}:[0-9]{2})?/, m)) {
    ts = m[0]
    sub(ts, dim ts reset, line)
  }

  # Levels
  gsub(/\bFATAL\b/, bred "FATAL" reset, line)
  gsub(/\bERROR\b/, red  "ERROR" reset, line)
  gsub(/\bWARN(ING)?\b/, yel  "&"   reset, line)
  gsub(/\bINFO\b/,  grn  "INFO" reset, line)
  gsub(/\bDEBUG\b/, cyn  "DEBUG" reset, line)
  gsub(/\bTRACE\b/, mag  "TRACE" reset, line)

  # Thread-like [ ... ] (single-pass; no infinite loop)
  gsub(/\[[^]]+\]/, blu "&" reset, line)

  # Logger-ish package.Class tokens (heuristic)
  gsub(/\b[A-Za-z_][A-Za-z0-9_$]*\.[A-Za-z0-9_$]+\.[A-Za-z0-9_$.]*[A-Za-z0-9_$]\b/, bblu "&" reset, line)

  print line
}
' "${1:-/dev/stdin}"
