#!/usr/bin/env bash
# log4jcat - colorize Log4j-style plain text logs using sed (no installs)
# Usage:
#   log4jcat file.log
#   cat file.log | log4jcat
#   log4jcat file.log | less -R

ESC=$'\033'
RESET="${ESC}[0m"

DIM="${ESC}[90m"
RED="${ESC}[31m"
BRED="${ESC}[1;31m"
YEL="${ESC}[33m"
GRN="${ESC}[32m"
CYN="${ESC}[36m"
MAG="${ESC}[35m"
DBLU="${ESC}[34m"       # dark-ish blue
BBLU="${ESC}[1;34m"

# Feed file or stdin
input="${1:-/dev/stdin}"

sed -E \
  -e "s/^([0-9]{4}-[0-9]{2}-[0-9]{2}([ T])[0-9]{2}:[0-9]{2}:[0-9]{2}([,\.][0-9]+)?(Z|[+-][0-9]{2}:[0-9]{2})?)/${DIM}\1${RESET}/" \
  -e "s/^([[:space:]]*Caused by:.*)/${BRED}\1${RESET}/" \
  -e "s/^([[:space:]]*Suppressed:.*)/${YEL}\1${RESET}/" \
  -e "s/^([[:space:]]+at[[:space:]].*)/${DIM}\1${RESET}/" \
  -e "s/^([[:space:]]*\\.\\.\\.[[:space:]]+[0-9]+[[:space:]]+more[[:space:]]*)$/${DIM}\1${RESET}/" \
  -e "s/\\[[^]]+\\]/${DBLU}&${RESET}/g" \
  -e "s/(^|[^A-Z])(FATAL)([^A-Z]|$)/\\1${BRED}\\2${RESET}\\3/g" \
  -e "s/(^|[^A-Z])(ERROR)([^A-Z]|$)/\\1${RED}\\2${RESET}\\3/g" \
  -e "s/(^|[^A-Z])(WARN|WARNING)([^A-Z]|$)/\\1${YEL}\\2${RESET}\\3/g" \
  -e "s/(^|[^A-Z])(INFO)([^A-Z]|$)/\\1${GRN}\\2${RESET}\\3/g" \
  -e "s/(^|[^A-Z])(DEBUG)([^A-Z]|$)/\\1${CYN}\\2${RESET}\\3/g" \
  -e "s/(^|[^A-Z])(TRACE)([^A-Z]|$)/\\1${MAG}\\2${RESET}\\3/g" \
  "$input"
